using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace b9
{
    public class Class1
    {
        private string izvodjac, album, zanr, godina, izdavackaKuca, slika;
        public string Izvodjac
        { get { return izvodjac; } }

        public string Album
        { get { return album; } }

        public string Zanr
        { get { return zanr; } }

        public string Godina
        { get { return godina; } }

        public string IzdavackaKuca
        { get { return izdavackaKuca; } }

        public string Slika
        { get { return slika; } }

        public void metoda(string s)
        {
            string[] podaci = s.Split('|');
            izvodjac = podaci[0];
            album = podaci[1];
            zanr = podaci[2];
            godina = podaci[3];
            izdavackaKuca = podaci[4];
            slika = podaci[5];
        }
    }
}
